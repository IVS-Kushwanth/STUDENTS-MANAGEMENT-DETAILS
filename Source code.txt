/*-----------------------DISPLAYING STUDENTS----------------------------------*/
void displayStudent() {
    FILE *fp = fopen(STUDENT_FILE, "r");
    struct Student st;

    if (!fp) {
        printf("Error opening file!\n");
        return;
    }

    printf("\n--- Student Records ---\n");
    while (fscanf(fp, "%d %s %f", &st.roll, st.name, &st.marks) == 3) {
        printf("Roll: %d | Name: %s | Marks: %.2f\n", st.roll, st.name, st.marks);
    }

    fclose(fp);
}
/*------------------------------SEARCH THE STUDENT RECORD--------------------*/
void searchStudent() {
    FILE *fp = fopen(STUDENT_FILE, "r");
    struct Student st;
    int roll, found = 0;

    if (!fp) {
        printf("Error opening file!\n");
        return;
    }

    printf("Enter Roll to search: ");
    scanf("%d", &roll);

    while (fscanf(fp, "%d %s %f", &st.roll, st.name, &st.marks) == 3)
    {
        if (st.roll == roll)
              {
            printf("Found: Roll: %d | Name: %s | Marks: %.2f\n", st.roll, st.name, st.marks);
            found = 1;
            break;
              }
 }

    if (!found) 
    {
        printf("Student not found!\n");
    }

    fclose(fp);
}
